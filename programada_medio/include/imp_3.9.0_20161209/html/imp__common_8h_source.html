156 static inline int calc_pic_size(int width, int height, IMPPixelFormat imp_pixfmt)
157 {
158  int bpp1 = 0, bpp2 = 1,size;
159 
160 #define BPP(FMT, A, B) case FMT: bpp1 = A;bpp2 = B;break
161  switch (imp_pixfmt) {
162  BPP(PIX_FMT_NV12, 3, 2);
163  BPP(PIX_FMT_YUYV422, 2, 1);
164  BPP(PIX_FMT_UYVY422, 2, 1);
165  BPP(PIX_FMT_RGB565BE, 2, 1);
166  default: break;
167  }
168 #undef BPP
169  size = width * height * bpp1 / bpp2;
170 
171  return size;
172 }
